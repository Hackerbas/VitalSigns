const Lab = ({ apiKey }) => {
  const [symptomInput, setSymptomInput] = useState('');
  const [diagnosisResult, setDiagnosisResult] = useState(null);
  const [isLoading, setIsLoading] = useState(false);
  
  // Calculator States
  const [weight, setWeight] = useState('');
  const [height, setHeight] = useState('');
  const [bmi, setBmi] = useState(null);
  const [waterWeight, setWaterWeight] = useState('');
  const [activityLevel, setActivityLevel] = useState('low'); 
  const [waterResult, setWaterResult] = useState(null);

  const handleDiagnosis = async (e) => {
    e.preventDefault();
    if (!symptomInput) return;
    setIsLoading(true);
    setDiagnosisResult(null);

    try {
      if (!apiKey || apiKey === "gsk_YOUR_ACTUAL_KEY_GOES_HERE") {
        throw new Error("No Key"); // Trigger simulation mode
      }

      const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
        method: "POST",
        headers: { "Authorization": `Bearer ${apiKey}`, "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "llama-3.1-8b-instant",
          messages: [
            { role: "system", content: `You are a helpful School Nurse Assistant. Analyze symptoms and return JSON ONLY: { "condition": "Short Name", "severity": "Low Risk/Moderate/URGENT", "color": "green/orange/red", "desc": "1 sentence explanation", "tips": "2-3 short steps", "action": "Next step" }.` },
            { role: "user", content: `Symptoms: ${symptomInput}` }
          ],
          response_format: { type: "json_object" }
        })
      });
      const data = await response.json();
      setDiagnosisResult(JSON.parse(data.choices[0].message.content));
    } catch (error) {
      // --- DEMO MODE FIX ---
      // If the API fails or no key is present, show a "Simulation" so the site looks cool for the club
      setTimeout(() => {
        setDiagnosisResult({ 
            condition: "Simulation: Common Cold", 
            severity: "Low Risk", 
            color: "green", 
            desc: "Based on the simulation, this appears to be a mild viral infection.", 
            tips: "Rest, drink water, and monitor temperature.", 
            action: "Visit the school nurse if fever persists." 
        });
      }, 1500); // Fake loading delay
    } finally { 
        // In real API calls we turn off loading here, but for the timeout above we need to wait
        if (apiKey && apiKey !== "gsk_YOUR_ACTUAL_KEY_GOES_HERE") {
            setIsLoading(false); 
        } else {
            setTimeout(() => setIsLoading(false), 1500);
        }
    }
  };

  const calculateBMI = (e) => {
    e.preventDefault();
    if (weight && height) setBmi((weight / ((height/100)**2)).toFixed(1));
  };

  const calculateWater = (e) => {
    e.preventDefault();
    let base = waterWeight * 35;
    if (activityLevel === 'medium') base += 500;
    if (activityLevel === 'high') base += 1000;
    setWaterResult((base / 1000).toFixed(2));
  };

  return (
    <div className="pt-32 pb-20 max-w-7xl mx-auto px-4">
      <div className="text-center mb-12">
        <h1 className="text-4xl font-bold text-white">AI Lab & Tools</h1>
        <p className="text-blue-200">Interactive health analysis powered by Groq AI.</p>
      </div>

      <div className="grid lg:grid-cols-2 gap-8">
        {/* AI NURSE */}
        <div className="bg-gradient-to-br from-slate-900 to-blue-950 border border-blue-500/30 rounded-3xl p-8 shadow-2xl">
           <div className="flex items-center gap-3 mb-6">
             <div className="bg-blue-600 p-2 rounded-lg"><Brain className="w-6 h-6 text-white" /></div>
             <h3 className="text-2xl font-bold text-white">AI Nurse Agent</h3>
           </div>
           <form onSubmit={handleDiagnosis} className="flex gap-4 mb-6">
             <input type="text" value={symptomInput} onChange={e=>setSymptomInput(e.target.value)} placeholder="Describe symptoms (e.g., headache, fever)..." className="w-full bg-blue-900/50 border border-blue-700 rounded-xl px-4 py-3 text-white focus:border-blue-400 outline-none" />
             <button disabled={isLoading} className="bg-blue-500 px-6 rounded-xl font-bold text-white hover:bg-blue-600 transition flex items-center">{isLoading ? <Loader2 className="animate-spin" /> : "Scan"}</button>
           </form>
           {diagnosisResult && (
             <div className="bg-white/5 border border-white/10 rounded-xl p-6 animate-fade-in">
               <div className="flex justify-between mb-2">
                 <h4 className="text-xl font-bold text-white">{diagnosisResult.condition}</h4>
                 <span className={`text-${diagnosisResult.color === 'red' ? 'red' : diagnosisResult.color === 'orange' ? 'orange' : 'green'}-400 font-bold uppercase text-sm`}>{diagnosisResult.severity}</span>
               </div>
               <p className="text-blue-200 text-sm mb-4">{diagnosisResult.desc}</p>
               <div className="bg-blue-950/50 p-3 rounded-lg text-sm text-slate-300 mb-2"><strong>Tips:</strong> {diagnosisResult.tips}</div>
               <div className="bg-blue-950/50 p-3 rounded-lg text-sm text-white"><strong>Action:</strong> {diagnosisResult.action}</div>
             </div>
           )}
        </div>

        {/* CALCULATORS */}
        <div className="space-y-6">
           {/* BMI */}
           <div className="bg-white rounded-3xl p-6 text-blue-950">
             <h3 className="text-xl font-bold mb-4 flex items-center gap-2"><Activity className="w-5 h-5 text-blue-600" /> BMI Calculator</h3>
             <div className="flex gap-2 mb-4">
               <input type="number" placeholder="Weight (kg)" onChange={e=>setWeight(e.target.value)} className="w-full bg-slate-100 p-3 rounded-xl" />
               <input type="number" placeholder="Height (cm)" onChange={e=>setHeight(e.target.value)} className="w-full bg-slate-100 p-3 rounded-xl" />
             </div>
             <button onClick={calculateBMI} className="w-full bg-blue-600 text-white font-bold py-3 rounded-xl">Calculate</button>
             {bmi && <div className="mt-4 text-center text-4xl font-black text-blue-900">{bmi}</div>}
           </div>

           {/* Water */}
           <div className="bg-blue-900 border border-white/10 rounded-3xl p-6">
             <h3 className="text-xl font-bold text-white mb-4 flex items-center gap-2"><Droplets className="w-5 h-5 text-cyan-400" /> Hydration</h3>
             <div className="flex gap-2 mb-4">
               <input type="number" placeholder="Weight (kg)" onChange={e=>setWaterWeight(e.target.value)} className="w-full bg-blue-950 p-3 rounded-xl text-white border border-blue-800" />
               <div className="flex gap-1">
                  {['low','medium','high'].map(l=><button key={l} onClick={()=>setActivityLevel(l)} className={`px-2 rounded-lg text-xs capitalize ${activityLevel===l?'bg-cyan-500 text-white':'bg-blue-950 text-blue-400'}`}>{l}</button>)}
               </div>
             </div>
             <button onClick={calculateWater} className="w-full bg-white text-blue-900 font-bold py-3 rounded-xl">Calculate</button>
             {waterResult && <div className="mt-4 text-center text-4xl font-black text-white">{waterResult}L</div>}
           </div>
        </div>
      </div>
    </div>
  );
};

                 <div className="flex items-center gap-2 mb-2">
                   {post.type === 'suggestion' ? (
                     <span className="bg-indigo-500/20 text-indigo-300 text-xs font-bold px-2 py-1 rounded flex items-center gap-1"><Lightbulb className="w-3 h-3" /> Suggestion</span>
                   ) : (
                     <span className="bg-purple-500/20 text-purple-300 text-xs font-bold px-2 py-1 rounded flex items-center gap-1"><Zap className="w-3 h-3" /> Project Update</span>
                   )}
                   <span className="text-slate-500 text-xs">{post.time}</span>
                   {/* Moved Votes to top right so you can still see popularity */}
                   <span className="ml-auto text-xs font-bold text-blue-400 flex items-center gap-1"><ThumbsUp className="w-3 h-3" /> {post.votes}</span>
                 </div>
                 <p className="text-white leading-relaxed text-lg">{post.content}</p>
                 
                 {/* Action Bar - FIXED */}
                 <div className="flex gap-4 mt-4 pt-4 border-t border-white/5 opacity-50 group-hover:opacity-100 transition">
                    <button onClick={() => handleVote(post.id, 1)} className="text-xs font-bold text-blue-300 hover:text-white flex items-center gap-1 hover:bg-white/5 px-2 py-1 rounded transition">
                        <ThumbsUp className="w-3 h-3" /> Bump
                    </button>
                    <button onClick={handleComment} className="text-xs font-bold text-blue-300 hover:text-white flex items-center gap-1 hover:bg-white/5 px-2 py-1 rounded transition">
                        <MessageSquare className="w-3 h-3" /> Comment
                    </button>
                 </div>
               </div>
             </div>
           ))}
        </div>

      </div>
    </div>
  );
};
